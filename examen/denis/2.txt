#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    FILE *fp;
    char linea[256];
    int c = 0;

    fp = popen("ps -e -o user", "r");
    if (fp == NULL) return 1;

    while (fgets(linea, sizeof(linea), fp)) {
        if (strstr(linea, "root"))
            c++;
    }

    printf("Root ha lanzado %d procesos\n", c - 1);

    pclose(fp);
    return 0;
}
