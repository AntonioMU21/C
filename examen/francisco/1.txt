#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp;
    char linea[256];
    int contador = 0;

    fp = popen("ss -tuln", "r");
    if (fp == NULL) {
        perror("Error al ejecutar ss");
        return 1;
    }

    while (fgets(linea, sizeof(linea), fp)) {
        if (strncmp(linea, "tcp", 3) == 0) {
            contador++;
        }
    }

    pclose(fp);

    printf("Puertos TCP abiertos: %d\n", contador);
    return 0;
}
