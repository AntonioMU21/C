#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    FILE *fp;
    char linea[256];
    int contador = 0;

    fp = popen("ps -e -o user", "r");
    if (fp == NULL) {
        printf("Error\n");
        return 1;
    }

    while (fgets(linea, sizeof(linea), fp) != NULL) {
        if (strstr(linea, "root") != NULL) {
            contador++;
        }
    }

    printf("Procesos de root: %d\n", contador - 1);

    pclose(fp);
    return 0;
}
